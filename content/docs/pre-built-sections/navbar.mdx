## Usage
Import the Button component and use it in your React application like so:

```tsx
import { Button } from 'fumadocs-ui/components/button';

function App() {
  return (
    <Button onClick={() => alert('Button clicked!')}>
      Click Me
    </Button>
  );
}
```
